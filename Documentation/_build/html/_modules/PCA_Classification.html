[docs]def zero_mix(x, n):
    """
    
     
    This function is used to padding,if the dimension is not equal.
    After all dimension is equal

    Parameters
    ----------
    x : (int)
        If 'x' is replacment variable.
    n : (int)
        If 'n' is noise of data.

    Returns
    -------
    temp : (int)
        It is return value of noise . If  less data presnt ,so it has uniformely added zeros in random location.
    
    
    """
    temp = np.copy(x)
    noise=n
    if 'spilt' in noise:
        frac = float(noise.split('-')[1])
    for i in temp:
        n = np.random.choice(len(i), int(round(frac * len(i))), replace=False)
        i[n] = 0
    return (temp)



[docs]def gaussian_mix(x):
    """
    
     
    This function is used to normalize. Mainly it creating a random sample from Gaussian distribution than normalize. 

    Parameters
    ----------
    x : (int)
        If 'x' is value which you want to normalize.

    Returns
    -------
    TYPE:(int)
        It has return the normalize value.

   

    
    
    """
    n = np.random.normal(0, 0.1, (len(x), len(x[0])))
    return (x + n)

[docs]       def accuracy_score(y_test,y_pred):
        """
         
        
        This function is used to calculate Accuracy,F1 Score Weighted,Recall Score Weighted,Precision Weighted,AUC Score ,MCC Score,Kappa,Log Loss.

        Parameters
        ----------
        y_test :(object) 
        This data has category labels for your test data, these labels will be used to test the accuracy between actual and predicted categories.
        y_pred : (object)
        This data has category labels for your predict data, these labels will be used to predict the accuracy between actual and predicted categories.

        Returns
        -------
        It doesn't anything return , It only print the value of  Accuracy,F1 Score Weighted,Recall Score Weighted,Precision Weighted,AUC Score ,MCC Score,Kappa,Log Loss. 
       
        
        """
        a=0
        for i in range(len(y_pred)):
          if(y_pred[i]==y_test[i]):
            a+=1
        return a*100/len(y_test)
